#include <stdio.h>

int main() {
    int n = 4;
    int size = 2 * n - 1;
    int i, j;

    for (i = 0; i < size; i++) {
        for (j = 0; j < size; j++) {
            int top = i;
            int left = j;
            int right = size - 1 - j;
            int bottom = size - 1 - i;

            int min = top;
            if (left < min) min = left;
            if (right < min) min = right;
            if (bottom < min) min = bottom;

            printf("%d ", n - min);
        }
        printf("\n");
    }

    return 0;
}
